            } else if (filter === 'oldest') {
                filtered.sort((a, b) => new Date(a.date) - new Date(b.date));

        // Download simulation
        function downloadFile(filename, type) {
            // Create a blob with dummy content
            const content = `// ${filename}\n// Downloaded from Sihas ESP Portfolio\n// Type: ${type}\n\n// This is a simulated download for demonstration purposes.`;
            const blob = new Blob([content], { type: 'text/plain' });
            const url = URL.createObjectURL(blob);
            
            const a = document.createElement('a');
            a.href = url;
            a.download = filename;
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
            URL.revokeObjectURL(url);
            
            // Show notification
            showNotification(`Downloading ${filename}...`);
        }
        
        // Add Project Modal
        function openAddProjectModal() {
            document.getElementById('add-project-modal').classList.remove('hidden');

            document.getElementById('add-project-modal').classList.add('hidden');

        function handleAddProject(e) {
            e.preventDefault();
            const form = e.target;
            const formData = new FormData(form);
            
            const newProject = {
                id: projects.length > 0 ? Math.max(...projects.map(p => p.id)) + 1 : 1,
                title: formData.get('title'),
                description: formData.get('description'),
                youtube: formData.get('youtube'),
                thumbnail: formData.get('thumbnail') || 'https://images.unsplash.com/photo-1518770660439-4636190af475?w=400',
                date: new Date().toISOString().split('T')[0],
                files: {
                    code: formData.get('file_code')?.name || null,
                    diagram: formData.get('file_diagram')?.name || null,
                    apk: formData.get('file_apk')?.name || null,
                    extra: formData.get('file_extra')?.name || null
                }
            };
            
            projects.unshift(newProject);
            saveProjects();
            closeAddProjectModal();
            form.reset();
            showNotification('Project added successfully!');
            
            if (!document.getElementById('projects-page').classList.contains('hidden')) {
                renderProjects();
            }
        }
        
        function deleteProject(id) {
            if (confirm('Are you sure you want to delete this project?')) {
                projects = projects.filter(p => p.id !== id);
                saveProjects();
                showNotification('Project deleted successfully!');
                showPage('projects');
            }
        }

        // Particle Animation
        function createParticles() {
            const container = document.getElementById('particles-container');
            const particleCount = 50;
            
            for (let i = 0; i < particleCount; i++) {
                const particle = document.createElement('div');
                particle.className = 'particle';
                particle.style.left = Math.random() * 100 + '%';
                particle.style.top = Math.random() * 100 + '%';
                particle.style.animationDuration = (Math.random() * 3 + 2) + 's';
                particle.style.animationDelay = Math.random() * 2 + 's';
                particle.style.opacity = Math.random() * 0.5 + 0.1;
                
                // Add floating animation
                particle.style.animation = `float ${Math.random() * 10 + 5}s ease-in-out infinite`;
                
                container.appendChild(particle);
            }
        }
